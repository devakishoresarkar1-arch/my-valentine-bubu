<script>
    const question = document.getElementById('question');
    const mainGif = document.getElementById('mainGif');
    const yesBtn = document.getElementById('yesBtn');
    const noBtn = document.getElementById('noBtn');
    const music = document.getElementById('bgMusic');

    let noCount = 0;
    let musicStarted = false;

    const messages = [
        "Will you be my Valentine? üíñ",
        "Wait... are you sure? ü•∫",
        "Think about it again! Please? üò≠",
        "But look how cute we are! ‚ù§Ô∏è‚Äçü©π",
        "Don't do this to me... üíî",
        "I'm taking the 'No' away now! Say YES! üò§"
    ];

    function startMusic() {
        if (!musicStarted) {
            music.play().catch(() => console.log("Music play blocked until interaction"));
            musicStarted = true;
        }
    }

    function createHeart() {
        const heart = document.createElement('div');
        heart.classList.add('heart');
        heart.innerHTML = '‚ù§Ô∏è';
        heart.style.left = Math.random() * 100 + 'vw';
        heart.style.animationDuration = Math.random() * 2 + 3 + 's';
        heart.style.opacity = Math.random();
        heart.style.fontSize = Math.random() * 20 + 10 + 'px';
        document.body.appendChild(heart);
        
        setTimeout(() => heart.remove(), 5000);
    }

    noBtn.addEventListener('click', () => {
        startMusic();
        noCount++;

        if (noCount < messages.length - 1) {
            question.innerText = messages[noCount];
            const currentSize = parseFloat(window.getComputedStyle(yesBtn).fontSize);
            yesBtn.style.fontSize = (currentSize * 1.3) + 'px';
            yesBtn.style.padding = (currentSize * 0.8) + 'px ' + (currentSize * 1.6) + 'px';
        } else {
            question.innerText = messages[messages.length - 1];
            noBtn.classList.add('hidden');
        }
    });

    yesBtn.addEventListener('click', () => {
        startMusic();
        document.body.style.backgroundColor = "#ffb3c1";
        question.innerText = "YAY! I KNEW IT! ü•∞‚ù§Ô∏è";
        mainGif.src = "https://media.tenor.com/m/v6aO3MZfVPUAAAAC/bubu-dudu-bubu-and-dudu.gif"; 
        
        noBtn.classList.add('hidden');
        yesBtn.classList.add('hidden');

        // Start the Heart Rain!
        setInterval(createHeart, 150);
    });
